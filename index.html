วิธีเชื่อมต่อ Google Sheets + Line Notify

ระบบนี้ส่งข้อมูลไป Google Apps Script ซึ่งทำหน้าที่เป็น backend กลาง — เขียนใบเข้า Sheets, แจ้ง Line, เพิ่ม Calendar โดยอัตโนมัติ ไม่มีค่าใช้จ่าย

1
สร้าง Google Sheet ชื่อ "VF Laser Quotations"
เปิด Google Sheets สร้าง spreadsheet ใหม่ — ชื่อใดก็ได้ จากนั้น copy Spreadsheet ID จาก URL (ส่วนที่อยู่ระหว่าง /d/ และ /edit)
2
เปิด Apps Script — วาง Code ด้านล่าง
ใน Google Sheet → Extensions → Apps Script → วาง code ด้านล่างทั้งหมด แล้วกด Save
// =================================================== // VISION FOLD DESIGN — Apps Script Backend // Google Sheets + Line Notify + Google Calendar // =================================================== const SHEET_NAME = "Quotations"; const LINE_NOTIFY_TOKEN = "YOUR_LINE_NOTIFY_TOKEN_HERE"; const CALENDAR_ID = "YOUR_CALENDAR_ID_HERE"; // หรือ "primary" function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow([ "เลขที่", "วันที่", "ลูกค้า", "ผู้ติดต่อ", "โทร", "ช่องทาง", "โปรเจกต์", "วัสดุ", "มูลค่า(฿)", "สถานะ", "กำหนดส่ง", "หมายเหตุ" ]); } saveToSheet(sheet, data); sendLineNotify(data); if (data.deliveryDate) addCalendarEvent(data); return ContentService .createTextOutput(JSON.stringify({success: true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function saveToSheet(sheet, d) { const rows = sheet.getDataRange().getValues(); const existingRow = rows.findIndex(r => r[0] === d.number); const row = [ d.number, d.quoteDate, d.clientName, d.clientContact, d.clientPhone, d.clientSource, d.projectName, d.material, d.total, d.status, d.deliveryDate, d.notes ]; if (existingRow > 0) { sheet.getRange(existingRow + 1, 1, 1, row.length).setValues([row]); } else { sheet.appendRow(row); } } function sendLineNotify(d) { if (!LINE_NOTIFY_TOKEN || LINE_NOTIFY_TOKEN === "YOUR_LINE_NOTIFY_TOKEN_HERE") return; const emoji = { draft: "", sent: "", approved: "", rejected: "", in_production: "", delivered: "" }[d.status] || ""; const msg = `\n${emoji} [${d.status.toUpperCase()}] ${d.number}\n` + ` ${d.clientName} (${d.clientSource})\n` + ` ${d.projectName}\n` + ` ฿${Number(d.total).toLocaleString()}\n` + (d.deliveryDate ? ` ส่งงาน: ${d.deliveryDate}` : ""); UrlFetchApp.fetch("https://notify-api.line.me/api/notify", { method: "post", headers: {Authorization: "Bearer " + LINE_NOTIFY_TOKEN}, payload: {message: msg} }); } function addCalendarEvent(d) { if (!CALENDAR_ID || CALENDAR_ID === "YOUR_CALENDAR_ID_HERE") return; const cal = CalendarApp.getCalendarById(CALENDAR_ID) || CalendarApp.getDefaultCalendar(); const date = new Date(d.deliveryDate); cal.createAllDayEvent( `Laser: ${d.clientName} — ${d.projectName}`, date, {description: `${d.number}\nมูลค่า: ฿${Number(d.total).toLocaleString()}\nหมายเหตุ: ${d.notes}`} ); }
3
Deploy เป็น Web App แล้วนำ URL มาใส่ที่นี่
Deploy → New deployment → Web app → Execute as: Me, Who has access: Anyone → Copy URL
4
รับ Line Notify Token
ไปที่ notify-bot.line.me/th → เลือก Group ที่ต้องการรับแจ้งเตือน → Generate Token → นำไปใส่ใน Apps Script บรรทัด LINE_NOTIFY_TOKEN
5
ทดสอบระบบ
กลับไปหน้า "สร้างใบ" → กรอกข้อมูล → กดปุ่ม "ส่งไป Sheets" → ตรวจสอบ Google Sheet และ Line Group
